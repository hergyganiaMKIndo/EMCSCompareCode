  SELECT t0.id ,
         t0.idcipl ,
         t0.idreference ,
         (
                SELECT
                       CASE
                              WHEN t0.referenceno = '-' THEN t0.casenumber
                              ELSE t0.referenceno
                       END) AS referenceno ,
         t0.idcustomer ,
         t0.NAME ,
         t0.uom AS unituom ,
         t0.partnumber ,
         t0.sn ,

         t0.quantity ,

                            ELSE t0.referenceno
                     END) AS referenceno ,
       t0.idcustomer ,
       t0.name ,
       t0.uom AS unituom ,
       t0.partnumber ,
       t0.sn ,

       t0.claim ,
